\documentclass{standalone}
\usepackage{pgfplots}
\pgfplotsset{
  compat=1.18, 
  trig format=rad, 
  ticklabel style = {font=\footnotesize},
}

\begin{document}
  \begin{tikzpicture}
    \begin{axis}[
      axis lines=middle,
      ymin=-1.5, ymax=1.5,
      xmin=-4*pi-pi/8, xmax=4*pi+pi/8,
      xlabel={\footnotesize \(x\)},
      xlabel style={at={(ticklabel* cs:1)}, anchor=west},
      ylabel={\footnotesize \(y\)},
      ylabel style={at={(ticklabel* cs:1)}, anchor=south},
      axis line style = {very thin},
      yticklabels={,,},
      xticklabels={,,},
      smooth,
      thick,
      no markers,
      tickwidth={1pt},
      ]
      \addplot[smooth, domain=0:2*pi] {cos(x)};
      \addplot[smooth, domain=-4*pi:0, dashed] {cos(x)};
      \addplot[smooth, domain=2*pi:4*pi, dashed] {cos(x)};
      \node[below] at (3*pi,-1) {\footnotesize \(y = \cos(x)\)};
    \end{axis}
  \end{tikzpicture}
\end{document}
