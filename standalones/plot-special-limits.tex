\pgfplotsset{special limit/.style={
    basic curve,
    enlargelimits,
    xtick={\empty}, ytick={\empty},
    xlabel={\empty}, ylabel={\empty},
}}

\begin{document}
\begin{tikzpicture}
  \begin{axis}[
    special limit,
    title={the ``magic'' limit for \(\sin(x)\) at \(0\)},
    ]
    \addplot[very thick, domain=-pi:pi] { sin(x) };
    \addplot[magenta, thick] coordinates { (-1,-1) (1,1) };
    \fill (axis cs:0,0) circle (2pt);
  \end{axis}
\end{tikzpicture}
