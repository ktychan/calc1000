\pgfplotsset{
  trigs/.style = {
    basic curve,
    enlargelimits,
    xlabel={\(\theta\)}, ylabel=\empty,
    grid=none,
    samples=500,
  },
  sincos/.style = {
    trigs,
    xmin={0}, xmax={pi/2},
    ymin={0}, ymax={1},
    xtick={ {pi/6}, {pi/4}, {pi/3}, {pi/2}},
    xticklabels={ \(\frac{\pi}{6}\), \(\frac{\pi}{4}\), \(\frac{\pi}{3}\), \(\frac{\pi}{2}\) },
    ytick={ {sqrt(2)/2}, {1/2}, {sqrt(3)/2}, {1} },
    yticklabels={ \(\sqrt{2}/2\), \(1/2\), \(\sqrt{3}/2\), \(1\) },
  },
  sincoslarge/.style = {
    trigs,
    width=5in,
    xmin={-pi}, xmax={3*pi},
    ymin={-1}, ymax={1},
    xtick={-pi,-pi/2,...,3*pi},
    xticklabels={,,0,,,,\(2\pi\),},
    ytick={-1,0,1},
    yticklabels={-1, 0, 1},
    domain={-2*pi}:{4*pi},
  }
}

\newcommand{\drawcircle}[2][]{
    \draw[dashed] (axis cs:0,0) circle (#2)
}

\newcommand{\drawtriangle}[3][]{
  \draw[very thick, #1] (axis cs:0,0) -- (axis cs:{(#3)*cos(#2)},0);
  \draw[very thick, #1] (axis cs:{(#3)*cos(#2)},0) -- (axis cs:{(#3)*cos(#2)},{(#3)*sin(#2)});
  \draw[very thick, #1] (axis cs:{(#3)*cos(#2)},{sin(#2)}) -- (axis cs:0,0);
}

\newcommand{\drawtheta}[4][]{
  \addplot[domain={0}:{#2}, #1] ({(#3)*cos(x)}, {(#3)*sin(x)});
  \node[right, #1] at (axis cs:{(#3)*cos(#2/2)}, {(#3)*sin(#2/2)}) {#4};
}
