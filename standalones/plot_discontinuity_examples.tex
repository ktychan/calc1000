\documentclass{standalone}
\usepackage{tikz}

\newcommand{\haxis}[3]{
  \draw[very thin, ->] (#1, 0) -- (#2, 0) node[right] {\footnotesize {#3}};
}
\newcommand{\htick}[2]{
    \draw[very thin] (#1, 0.05) -- (#1, -0.05) node[below] {\footnotesize #2};
  }
\newcommand{\hticks}[2]{
  \foreach \x in {#1,...,#2} { \htick{\x}{\(\x\)} }
}
\newcommand{\vaxis}[3]{
  \draw[very thin, ->] (0, #1) -- (0, #2) node[above] {\footnotesize #3};
}
\newcommand{\vtick}[2]{
  \draw[very thin] (0.05, #1) -- (-0.05, #1) node[left] {\footnotesize #2};
}
\newcommand{\vticks}[2]{
  \foreach \y in {#1,...,#2} { \vtick{\y}{\(\y\)} }
}

\begin{document}
\begin{tikzpicture}

  \begin{scope}
    \haxis{0}{4}{\(x\)}
    \hticks{1}{3}
    \vaxis{0}{4}{\(y\)}
    \vticks{1}{3}

    \draw[smooth, domain=0:3] plot ({\x}, {\x + 1}) node[below right] {\(g_{1}(x)\)};
    \draw[fill=white] (1,2) circle (0.05);
  \end{scope}

  \begin{scope}[shift={(7,0)}]
    \clip (-2.5,-1) rectangle (2.5,4.5);
    \haxis{-2}{2}{\(x\)}
    \hticks{-1}{1}
    \vaxis{0}{4}{\(y\)}
    \vticks{1}{3}
    % \vaxis{-4}{4}{\(y\)};
    % \vticks{1}{3}; 
    \draw[smooth, domain=-2:0] plot ({\x}, {\x*\x});
    \draw[smooth, domain=0:2] plot ({\x}, {\x + 1}) node[above] {\(g_{2}(x)\)};;
    \draw[fill] (0,0) circle (0.05);
    \draw[fill=white] (0,1) circle (0.05);
  \end{scope}

  \begin{scope}[shift={(9,2)},scale=0.5]
    \clip (2,-4) rectangle (8.5,4.5);
    \haxis{1}{8}{\(x\)}
    \hticks{3}{7};
    % \vaxis{-4}{4}{\(y\)};
    % \vticks{1}{3}; 
    \draw[dotted] (5,-10) -- (5,10);
    \draw[smooth, domain=1:4.9] plot ({\x}, {1/3*(\x*\x - 2*\x - 3)/(\x - 5)});
    \draw[smooth, domain=5.1:10] plot ({\x}, {1/10*(\x*\x - 2*\x - 3)/(\x-5)))});
    \node at (7, 3) {\(g_{3}(x)\)};
  \end{scope}
\end{tikzpicture}
\end{document}
