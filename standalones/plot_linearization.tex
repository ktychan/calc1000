\documentclass{standalone}
\usepackage{pgfplots,xcolor}
\pgfplotsset{compat=1.18}

% see https://tex.stackexchange.com/questions/18359/plotting-an-implicit-function-using-pgfplots
% for implicit plots using gnuplot
\begin{document}
\begin{tikzpicture}
  \begin{axis}[
    xlabel={\footnotesize \(x\)},
    ylabel={\footnotesize \(y\)},
    xmin={-0.5},
    xmax={5.5},
    ymin={-0.5},
    ymax={3.5},
    axis x line={middle},
    axis y line={middle},
    xlabel style={at={(ticklabel* cs:1)}, anchor=west},
    ylabel style={at={(ticklabel* cs:1)}, anchor=south},
    axis equal
    ]
  \addplot[domain=0:5, no markers, smooth, samples=1000]{sqrt(x)};
  \addplot[teal, domain=0:4, no markers] { x/2 + 1/2 };
  \draw[fill] (1,1) circle (0.05);
  \node[teal, above] at (3,2.5) {\footnotesize linearization of \(\sqrt{x}\) at \(a = 1\)};
  \node[below] at (4,2) {\footnotesize \(\sqrt{x}\)};
  \end{axis}
\end{tikzpicture}
\end{document}
