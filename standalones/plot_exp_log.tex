\documentclass{standalone}
\usepackage{pgfplots,xcolor}
\pgfplotsset{compat=1.18}

% see https://tex.stackexchange.com/questions/18359/plotting-an-implicit-function-using-pgfplots
% for implicit plots using gnuplot
\begin{document}
\begin{tikzpicture}
  \begin{axis}[
    xlabel={\footnotesize \(x\)},
    ylabel={\footnotesize \(y\)},
    xmin={-1.5},
    xmax={5.5},
    ymin={-1.5},
    ymax={5.5},
    axis x line={middle},
    axis y line={middle},
    xlabel style={at={(ticklabel* cs:1)}, anchor=west},
    ylabel style={at={(ticklabel* cs:1)}, anchor=south},
    yticklabels={,,},
    xticklabels={,,},
    domain=-1:5,
    samples=1000,
    ]
  \addplot[dotted] coordinates { (-5,-5) (5,5) };
  \addplot[smooth, thick, blue]{exp(x)};
  \addplot[smooth, thick, blue]{ln(x)};
  \node[blue, right] at (1.5,5) {};
  \node at (2,4) {\footnotesize \(e^{x}\)};
  \node at (4,2) {\footnotesize \(\ln(x)\)};
  \node[right] at (4,4) {\footnotesize \(x=y\)};
  \end{axis}
\end{tikzpicture}
\end{document}
